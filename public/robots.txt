# Block admin, APIs and user-state pages.
#
#   /download/  is intentionally NOT blocked here.
#   The /download/{id} pages already emit `<meta name="robots" content="noindex,follow">`
#   from generateMetadata(). Blocking them in robots.txt would prevent Googlebot
#   from ever reading that noindex tag, leaving the URLs stuck in
#   "Discovered – currently not indexed" forever (Search Console flagged
#   ~10k of these). Letting Googlebot crawl them once is the right behavior;
#   internal links use rel="nofollow" so the crawl budget stays small.
Disallow: /dashboard
Disallow: /dashboard/
Disallow: /api/
Disallow: /backend-api/
Disallow: /_next/
Disallow: /login
Disallow: /register
Disallow: /forgot-password
Disallow: /reset-password
Disallow: /verify-email
Disallow: /auth/
